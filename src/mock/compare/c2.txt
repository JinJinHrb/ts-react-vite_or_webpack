const featInfo = rawFeatInfo
  .map(el => {
    const name = _.trim(el.name)
    const rtn = {
      name,
      title: el.fieldJson?.title,
      component: el.fieldJson['x-component'],
      value: detail[name],
    } as any
    if (_.isArray(el.fieldJson.enum)) {
      rtn.dataSource = el.fieldJson.enum
      if (rtn.value) {
        const item = (rtn.dataSource || []).find((el: any) => el.value === rtn.value)
        if (item) {
          rtn.value = item?.label
        } else if (_.isArray(el.fieldJson.hideEnum)) {
          const hideItem = (el.fieldJson.hideEnum || []).find((el: any) => el.value === rtn.value)
          if (hideItem) {
            rtn.value = hideItem?.label
          }
        }
      }
    }
    return rtn
  })
  .filter(el => !['XtCustomSelect'].includes(el.component))
